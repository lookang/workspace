omega = np.sqrt((2.0*q*E/(d*m))) ## EDIT THIS ##

#########################################
# We can numerically solve ODEs using odeint functions from 
# scipy libraries
#########################################

def torque(x,t,q=1.0,d=1.0,m=1.0,E=100):
    '''
    This function is the right hand side of the ODE which
    describes the torque on 
    the dipole without the small angle approximation.
    the term (2.0*q*E/(d*m)) is just the constant C in the equation

    x'' = -C x

    INPUTS:
    x = vector containing [x,v] units [m,m/s]
    t = time (s)
    q = charge (C)
    d = distance (m)
    m = kilograms (kg)
    E = electric field (N/C)

    '''
    return [x[1],-(2.0*q*E/(d*m))*np.sin(x[0])]

############################################################
############################################################

from scipy.integrate import ode, odeint

#Set the times to evaluate 
t = np.linspace(0,Tmax,N)

#convert initial angle to radians
theta_rads = theta_degrees * np.pi / 180.0


##############################
# compute approximate theta values on the time intervals

##############################

x0 = [theta_rads, 0.0]

##############################
# numerically solve the real solution using odeint
sol2 = odeint(torque,x0,t,args=(q,d,m,E))

#############################
# Plot the two solutions
plt.plot(t,theta_rads*np.cos(omega*np.array(t)),label="small angle approx")
plt.plot(t,sol2[:,0],label='good numerical solution')
